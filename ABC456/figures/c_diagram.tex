\documentclass[tikz,border=10pt]{standalone}
\usepackage{ctex}
\usepackage{tikz}
\usetikzlibrary{arrows.meta, calc, positioning}

\definecolor{darktext}{RGB}{44,62,80}
\definecolor{extend}{RGB}{46,204,113}
\definecolor{breakc}{RGB}{231,76,60}

\begin{document}
\begin{tikzpicture}[font=\large, >=Stealth, line cap=round, line join=round]

% Title
\node[font=\bfseries\Large, text=darktext] at (5, 2.8) {C题：$f[i]$ = 以第 $i$ 个字符结尾的合法子串数};

% String
\foreach \x/\ch in {1/a, 2/b, 3/b, 4/c} {
    \node[draw, thick, fill=white, minimum width=1.2cm, minimum height=1.0cm, font=\bfseries] (s\x) at (\x*2, 1.5) {\ch};
    \node[below=2pt, font=\small, text=darktext] at (s\x.south) {\x};
}

% f values
\node[font=\bfseries, extend] (f1) at (2, 0.3) {$1$};
\node[font=\bfseries, extend] (f2) at (4, 0.3) {$2$};
\node[font=\bfseries, breakc] (f3) at (6, 0.3) {$1$};
\node[font=\bfseries, extend] (f4) at (8, 0.3) {$2$};

% Substrings
\node[below=1pt, font=\tiny, text=darktext] at (f1.south) {a};
\node[below=1pt, font=\tiny, text=darktext] at (f2.south) {b,ab};
\node[below=1pt, font=\tiny, text=darktext] at (f3.south) {b};
\node[below=1pt, font=\tiny, text=darktext] at (f4.south) {c,bc};

% Arrows
\draw[->, very thick, extend] (f1) -- (f2);
\draw[->, very thick, breakc, dashed] (f2) -- (f3);
\draw[->, very thick, extend] (f3) -- (f4);

% Formula
\node[below=0.4cm, font=\normalsize, text=darktext] at (f2.south) {
    $f[i]=1$（断裂）\quad 或 \quad $f[i]=f[i-1]+1$（延伸）
};

\end{tikzpicture}
